166 int DistanceBetweenTwoVertices(int from, int to) const override {
167 if (adj_list.find(from) == adj_list.end() || adj_list.find(to) == adj_list.end()) {
168 return 0;

170
171 std::unordered_map<int, bool> is_visited = GenerateVisitedMap();
172 std::unordered_map<int, int> distance;
173 for (const auto& p : adj_list) {
174 distance[p.first] = 0;
175 }
176
177 std::queue<int> queue;
178
179 queue.push(from);
180 is_visited[from] = true;
181 while (!queue.empty()) {
182 int current = queue.front();
183 queue.pop();
184
185 for (int adj_vertex : adj_list.at(current)) {
186 if (!is_visited[adj_vertex]) {
187 distance[adj_vertex] = distance[current] + 1;
188 queue.push(adj_vertex);

214
215 if (!is_visited[start]) {
216 is_visited[start] = true;
217 }
218
219 for (int adj_vertex : adj_list.at(start)) {
220 if (!is_visited[adj_vertex]) {
